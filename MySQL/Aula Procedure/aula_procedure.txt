(pesquisa_id INT)

DELIMITER //
CREATE PROCEDURE atualizar_estudante
(pesquisar_id INT, nova_matri INT, novo_curso VARCHAR(75))
BEGIN
    UPDATE estudantes
    SET matricula = nova_matri, curso = novo_curso
    WHERE id = pesquisar_id;
    SELECT * FROM estudantes
    WHERE id = pesquisar_id;
END //
